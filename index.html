FITHCore Компоненти

Core System

-- core/init.lua FITHCore = { version = '1.0.0', debug = false, modules = {}, cache = {}, config = {} } function FITHCore.Initialize() FITHCore.LoadConfig() FITHCore.InitializeDatabase() FITHCore.InitializeCache() FITHCore.InitializeSecurity() FITHCore.LoadModules() end function FITHCore.LoadModules() for _, module in ipairs(FITHCore.config.modules) do local success, result = pcall(function() return require(string.format('modules/%s/init', module)) end) if success then FITHCore.modules[module] = result else print(string.format('Failed to load module: %s', module)) end end end return FITHCore

Security System

-- security/anticheat.lua local AntiCheat = { checks = {}, violations = {}, thresholds = {} } function AntiCheat.Initialize() AntiCheat.LoadChecks() AntiCheat.StartMonitoring() end function AntiCheat.RegisterCheck(name, callback, threshold) AntiCheat.checks[name] = callback AntiCheat.thresholds[name] = threshold end function AntiCheat.HandleViolation(player, checkName, data) if not AntiCheat.violations[player] then AntiCheat.violations[player] = {} end table.insert(AntiCheat.violations[player], { check = checkName, time = os.time(), data = data }) if #AntiCheat.violations[player] >= AntiCheat.thresholds[checkName] then AntiCheat.PunishPlayer(player) end end return AntiCheat
